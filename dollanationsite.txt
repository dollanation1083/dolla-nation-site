         JavaScript for Mobile Menu Toggle
        document.addEventListener('DOMContentLoaded', () = {
            const button = document.getElementById('mobile-menu-button');
            const menu = document.getElementById('mobile-menu');

            button.addEventListener('click', () = {
                menu.classList.toggle('hidden');
            });

             Close menu when a link is clicked (for better mobile UX)
            menu.querySelectorAll('a').forEach(link = {
                link.addEventListener('click', () = {
                    menu.classList.add('hidden');
                });
            });

             Simple form submission handler (prevent actual submission for this demo)
            const form = document.querySelector('#contact form');
            form.addEventListener('submit', function(e) {
                e.preventDefault();
                
                 Show a friendly message instead of a generic alert()
                const messageContainer = document.createElement('div');
                messageContainer.className = 'fixed inset-0 bg-dark-bg80 backdrop-blur-sm z-[100] flex items-center justify-center';
                messageContainer.innerHTML = `
                    div class=bg-card-bg p-8 rounded-xl shadow-2xl border-t-4 border-primary max-w-sm text-center
                        h4 class=text-xl font-bold text-white mb-3Message Sent!h4
                        p class=text-gray-400 mb-6Thank you for reaching out. DOLLA NATION will get back to you soon.p
                        button id=close-message class=bg-primary text-white px-4 py-2 rounded-lg hoverbg-indigo-700 transition duration-200Closebutton
                    div
                `;
                document.body.appendChild(messageContainer);

                document.getElementById('close-message').addEventListener('click', () = {
                    document.body.removeChild(messageContainer);
                });

                 Clear the form
                form.reset();
            });
        });
